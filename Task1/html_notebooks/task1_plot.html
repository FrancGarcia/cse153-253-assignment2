In[37]:
# Bar width and positions
x = np.arange(len(INSTRUMENTS))
bar_width = 0.25

# Plot
fig, ax = plt.subplots()
ax.bar(x - bar_width, hmm_llh, bar_width, label='HMM')
ax.bar(x, markov_llh, bar_width, label='Markov Chain')
ax.bar(x + bar_width, lstm_quartet_llh, bar_width, label='LSTM-Quartet')

# Labels and formatting
ax.set_xlabel('Instruments')
ax.set_ylabel('Log-Likelihood')
ax.set_title('Model Log-Likelihood per Instrument')
ax.set_xticks(x)
ax.set_xticklabels(INSTRUMENTS)
ax.legend()
ax.grid(True, linestyle='--', alpha=0.5)
ax.invert_yaxis()  # Flip y-axis

plt.tight_layout()
plt.show()
